================
Repeat condition
================

// variable
5 => int x;

// in the repeat condition, 'x' is evaluated once and only once, at
// the start of the loop; if the value of x is updated subsequently,
// it will not alter the number of repetitions
repeat( x )
{
    // print now in seconds
    <<< now / second >>>;
    // advance time
    500::ms => now;
}
